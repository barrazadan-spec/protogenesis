  namespace Protogenesis.Views {
    public class ViewManager : MonoBehaviour {
      public static ViewManager Instance { get; private set; }
      public ViewType CurrentView { get; private set; } = ViewType.Interior;
   
      [SerializeField] Camera cameraInterior;
      [SerializeField] Camera cameraExterior;
   
      void Awake() { /* singleton */ }
   
      void Update() {
        if (Input.GetKeyDown(KeyCode.Tab))
          SwitchView();
      }
   
      public void SwitchView() {
        if (CurrentView == ViewType.Interior)
          SetView(ViewType.Exterior);
        else
          SetView(ViewType.Interior);
      }
   
      public void SetView(ViewType view) {
        CurrentView = view;
        cameraInterior.enabled = (view == ViewType.Interior);
        cameraExterior.enabled = (view == ViewType.Exterior);
        EventBus.OnViewChanged?.Invoke(view);
        Debug.Log($'[ViewManager] Vista: {view}');
      }
    }
  }
   
  PASO 4 — ASIGNAR EN LA ESCENA:
  Agregar ViewManager como componente de ViewSystemRoot.
  Asignar CameraInterior y CameraExterior en el Inspector.
   
  PASO 5 — PRUEBA:

  Verificar que Tab muestra uno u otro pero nunca los dos juntos.
